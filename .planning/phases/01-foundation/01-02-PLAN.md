    **`tests/auth/email-password.test.ts`:**
    ```typescript
    import { auth } from '@/lib/auth/auth';
    import { createTestUser } from '../helpers/test-db';

    describe('AUTH-02: email/password sign in', () => {
      it('returns a session for valid credentials', async () => {
        const { user } = await createTestUser({ email: 'test@example.com', password: 'password123' });
        const result = await auth.api.signInEmail({ body: { email: user.email, password: 'password123', rememberMe: false } });
        expect(result.session).toBeDefined();
        expect(result.user.email).toBe(user.email);
      });
      it('returns error for invalid password', async () => {
        const { user } = await createTestUser({ email: 'test2@example.com', password: 'correct' });
        await expect(auth.api.signInEmail({ body: { email: user.email, password: 'wrong' } })).rejects.toThrow();
      });
    });

    describe('AUTH-04: session persistence', () => {
      it('rehydrates session from cookie headers', async () => {
        const { session } = await createTestUser({ email: 'test3@example.com', password: 'pw123456' });
        const rehydrated = await auth.api.getSession({ headers: new Headers({ cookie: `better-auth.session_token=${session.token}` }) });
        expect(rehydrated?.user.email).toBe('test3@example.com');
      });
    });

    describe('AUTH-06: sign out', () => {
      it('clears the session cookie', async () => {
        const { session } = await createTestUser({ email: 'test4@example.com', password: 'pw123456' });
        await auth.api.signOut({ headers: new Headers({ cookie: `better-auth.session_token=${session.token}` }) });
        const result = await auth.api.getSession({ headers: new Headers({ cookie: `better-auth.session_token=${session.token}` }) });
        expect(result).toBeNull();
      });
    });
    ```
